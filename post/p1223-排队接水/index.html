P1223 排队接水

2019-05-10
#include <cstdio>

#include <cstring>

#include <algorithm>

#include <iostream>

using namespace std;

int n;

struct people{

    int bh;

    int sj;

}p[1005];

bool cmd(people a,people b){

    if(a.sj!=b.sj)return a.sj<b.sj;

    else return a.bh<b.bh;

}

int main(){

    scanf("%d",&n);

    for(int i=1;i<=n;i++){

        scanf("%d",&p[i].sj);

        p[i].bh=i;

    }

    sort(p+1,p+n+1,cmd);

    for(int i=1;i<=n;i++){printf("%d",p[i].bh);if(i+1<=n)printf(" ");}

    printf("\n");

    double sum;//wa于此，int sum然后double一个aa sum*1.0/n,找了二十分钟。。。

    for(int i=1;i<=n;i++){

        sum+=i*p[n-i].sj;

    }

    sum=sum/n;

    printf("%.2f",sum);

    return 0;

}

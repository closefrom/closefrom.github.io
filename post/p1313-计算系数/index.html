P1313 计算系数

2019-05-10

算法：二项式定理，数论，递推

要用到：组合数（杨辉三角），乘方

根据二项式定理，（x+y）k中xm*y^（k-m）的系数为C（k，m）

则有ax+by）k中xm_y（k-m）的系数为C（k，m）_am*b^（k-m）

流程：

1.输入参数

2.使用杨辉三角计算C（k，m）

3.计算am与b（k-m）

AC Code：
#include <cstdio>
#define ll long long
#define mod 10007
using namespace std;
int a,b,k,n,m,c[1005][1005];

ll ksm(int sa,int sb){
    if(sb==0)return 1;
    ll ans=1;
    for(int i=1;i<=sb;i++){
        ans*=sa;
        ans%=mod;
    }
    return ans;
}
int main(){
    scanf("%d%d%d%d%d",&a,&b,&k,&n,&m);
    c[0][0]=1;
    for(int i=1;i<=k;i++)c[i][0]=c[i][i]=1;
    for(int i=1;i<=k;i++)for(int j=1;j<i;j++)c[i][j]=(c[i-1][j-1]+c[i-1][j])%mod;
    ll as;
    as=c[k][n]*(ksm(a,n)*ksm(b,m)%mod)%mod;
    printf("%d",as);
    return 0;
}

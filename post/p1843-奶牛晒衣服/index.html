P1843 奶牛晒衣服

2019-05-10
#include <cstdio>
using namespace std;
int n,a,b,nn[500005];

int main(){
    scanf("%d%d%d",&n,&a,&b);
    long long l=0,r=0;
    for(int i=1;i<=n;i++){scanf("%d",&nn[i]);r+=nn[i];}
    while(l<=r){
        int sum=0;
        long long mid=(l+r)>>1;
        for(int i=1;i<=n;i++){
            if(nn[i]-a*mid>0)sum+=(nn[i]-a*mid)%b==0?(nn[i]-a*mid)/b:(nn[i]-a*mid)/b+1;
        }
        if(sum>mid)l=mid+1;
        else r=mid-1;
    }
    printf("%lld",l);
    return 0;   
}

P1306 斐波那契公约数(数论+矩阵快速幂)

2019-05-10




//答案在2*2矩阵的左下角
#include <cstdio>
#include <cstring>
#define ll long long
#define mod 100000000
using namespace std;
int n,m;
int gcd(int a,int b){return b?gcd(b,a%b):a;}
struct node{
    ll a[2][2];
    node(){memset(a,0,sizeof(a));}
};
node mul(node a,node b){
    node c;
    for(int i=0;i<2;i++){
        for(int j=0;j<2;j++){
            for(int k=0;k<2;k++){
                c.a[i][j]+=a.a[i][k]*b.a[k][j];
                c.a[i][j]%=mod;
                
            }
        }
    }
    return c;
}
node jzjc(node a,int x){
    node ans;
    for(int i=0;i<2;i++)ans.a[i][i]=1;
    while(x){
        if(x&1)ans=mul(ans,a);
        a=mul(a,a);
        x>>=1;
    }
    return ans;
}
int main(){
    scanf("%d%d",&n,&m);
    int sm=gcd(n,m);
    //printf("%d\n",sm);
    node fbi;
    fbi.a[0][0]=fbi.a[0][1]=fbi.a[1][0]=1;
    node as=jzjc(fbi,sm);
    //printf("%d %d\n%d %d\n\n",as.a[0][0],as.a[0][1],as.a[1][0],as.a[1][1]);
    printf("%lld",as.a[1][0]);
    return 0;
}

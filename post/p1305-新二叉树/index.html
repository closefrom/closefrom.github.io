P1305 新二叉树

2019-05-10
#include <cstdio>
#include <iostream>
using namespace std;
int n;
char c[30][3];
void ptree(char c2){
    if(c2!='*'){
        cout << c2;
        for(int i=1;i<=n;i++){
            if(c[i][0]==c2){
                ptree(c[i][1]);
                ptree(c[i][2]);
            }
        }
    }
    return;
}
int main(){
    cin >> n;
    for(int i=1;i<=n;i++)cin >> c[i][0] >> c[i][1] >> c[i][2];
    ptree(c[1][0]);
    return 0;
}
